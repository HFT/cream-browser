00067 static inline void luaL_callfunction (lua_State *L, int ref, int nargs, int nreturns)
00068 {
00069      if (ref)
00070      {
00071           int errfunc;
00072 
00073           lua_rawgeti (L, LUA_REGISTRYINDEX, ref);
00074 
00075           /* Move function before arguments */
00076           lua_insert (L, - nargs - 1);
00077 
00078           /* push error handler and move it before args and function */
00079           lua_pushcfunction (L, luaL_error_handler);
00080           lua_insert (L, - nargs - 2);
00081           errfunc = lua_gettop (L) - nargs - 1;
00082 
00083           if (lua_pcall (L, nargs, nreturns, - nargs - 2))
00084           {
00085                warn ("%s", lua_tostring (L, -1));
00086                lua_pop (L, 2);
00087                return;
00088           }
00089 
00090           lua_remove (L, errfunc);
00091      }
00092 }
